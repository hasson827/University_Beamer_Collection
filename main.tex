\section{简介}

\begin{frame}{什么是 Beamer}
  Beamer 是一个用于制作演示文稿的 \LaTeX{} 文档类，特别适合学术报告 \cite{beamer}。
  \begin{itemize}
    \item 专业的数学公式排版
    \item 稳定的交叉引用
    \item 与 \LaTeX{} 文档无缝集成
  \end{itemize}
\end{frame}

\begin{frame}{数学公式示例}
  著名的欧拉公式：
  \[
    e^{i\pi} + 1 = 0
  \]
  
  二次方程求根公式：
  \[
    x = \frac{-b \pm \sqrt{b^2 - 4ac}}{2a}
  \]
\end{frame}
